sentence = input("Enter sentence:")
sentence1 = sentence.split()
i = 0
while i < len(sentence1):
    a = sentence1[i]
    i = i+1
    count = 0
    j = 0
    while j < len(sentence1):
        if(a == sentence1[j]):
            count +=1
        j = j+1
    print("Word: ",a,"  ","Count: ", count)
sentence1 = set(sentence1)
print(sentence1)
